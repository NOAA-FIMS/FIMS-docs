221 void evaluate_age_comp() {
222 for (size_t y = 0; y < this->nyears; y++) {
223 Type sum = static_cast<Type>(0.0);
224 Type sum_obs = static_cast<Type>(0.0);
225 // robust_add is a small value to add to expected compostion
226 // proportions at age to stabilize likelihood calculations
227 // when the expected proportions are close to zero.
228 //Type robust_add = static_cast<Type>(0.0); // zeroed out before testing 0.0001;
229 // sum robust is used to calculate the total sum of robust
230 // additions to ensure that proportions sum to 1.
231 //Type robust_sum = static_cast<Type>(1.0);
232
233 for (size_t a = 0; a < this->nages; a++) {
234 size_t i_age_year = y * this->nages + a;
235 // Here we have a check to determine if the age comp
236 // should be calculated from the retained landings or
237 // the total population. These values are slightly different.
238 // In the future this will have more impact as we implement
239 // timing rather than everything occuring at the start of
240 // the year.
241 if(this->fleet_observed_landings_data_id_m == -999){
242 this->agecomp_expected[i_age_year] =
243 this->index_numbers_at_age[i_age_year];
244 }else {
245 this->agecomp_expected[i_age_year] =
246 this->landings_numbers_at_age[i_age_year];
247 }
248 sum += this->agecomp_expected[i_age_year];
249 //robust_sum -= robust_add;
250
251 // This sums over the observed age composition data so that
252 // the expected age composition can be rescaled to match the
253 // total number observed. The check for na values should not
254 // be needed as individual years should not have missing data.
255 // This is need to be re-explored if/when we modify FIMS to
256 // allow for composition bins that do not match the population
257 // bins.
258 if( this->fleet_observed_agecomp_data_id_m != -999) {
259 if( this->observed_agecomp_data->at(i_age_year)!=
260 this->observed_agecomp_data->na_value) {
261 sum_obs += this->observed_agecomp_data->at(i_age_year);
262 }
263 }
264 }
265 for (size_t a = 0; a < this->nages; a++) {
266 size_t i_age_year = y * this->nages + a;
267 this->agecomp_proportion[i_age_year] =
268 this->agecomp_expected[i_age_year] / sum;
269 //robust_add + robust_sum * this->agecomp_expected[i_age_year] / sum;
270
271 if( fleet_observed_agecomp_data_id_m != -999) {
272 this->agecomp_expected[i_age_year] =
273 this->agecomp_proportion[i_age_year] *
274 sum_obs;
