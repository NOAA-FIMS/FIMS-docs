233 void SetupPriors() {
234 for (density_components_iterator it = density_components.begin(); it != density_components.end(); ++it) {
235 std::shared_ptr<fims_distributions::DensityComponentBase<Type> > d = (*it).second;
236 if (d->input_type == "prior") {
237 FIMS_INFO_LOG("Setup prior for distribution " + fims::to_string(d->id));
238 variable_map_iterator vmit;
239 FIMS_INFO_LOG("Link prior from distribution " + fims::to_string(d->id) + " to parameter " + fims::to_string(d->key[0]));
240 d->priors.resize(d->key.size());
241 for (size_t i = 0; i < d->key.size(); i++) {
242 FIMS_INFO_LOG("Link prior from distribution " + fims::to_string(d->id)
243 + " to parameter " + fims::to_string(d->key[0]));
244 vmit = this->variable_map.find(d->key[i]);
245 d->priors[i] = (*vmit).second;
246 }
247 FIMS_INFO_LOG("Prior size for distribution " + fims::to_string(d->id) + "is: " + fims::to_string(d->x.size()));
248 }
249 }
