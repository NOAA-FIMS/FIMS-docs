283 JsonValue JsonParser::ParseNumber() {
284 size_t end_pos = position;
285 bool is_float = false;
286 while (end_pos < data.size() &&
287 (std::isdigit(data[end_pos]) || data[end_pos] == '.' ||
288 data[end_pos] == '-' || data[end_pos] == 'e' ||
289 data[end_pos] == 'E')) {
290 if (data[end_pos] == '.' || data[end_pos] == 'e' || data[end_pos] == 'E') {
291 is_float = true;
292 }
293 end_pos++;
294 }
295
296 std::string num_str = data.substr(position, end_pos - position);
297 position = end_pos;
298
299 if (is_float) {
300 double num;

302 return JsonValue(num);
