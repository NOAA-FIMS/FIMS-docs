398
399 for (size_t i = 0; i < log_init_naa.size(); i++) {

401 if (this->log_init_naa[i].estimated_m) {
402 info->RegisterParameterName("log_init_naa");
403 info->RegisterParameter(population->log_init_naa[i]);
