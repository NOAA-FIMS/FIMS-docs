28 bool CreateTMBModel() {
29  for (size_t i = 0; i < FIMSRcppInterfaceBase::fims_interface_objects.size();
30  i++) {
31  FIMSRcppInterfaceBase::fims_interface_objects[i]->add_to_fims_tmb();
32  }
33 
34  // base model
35  std::shared_ptr<fims_info::Information<TMB_FIMS_REAL_TYPE>> d0 =
36  fims_info::Information<TMB_FIMS_REAL_TYPE>::GetInstance();
37  d0->CreateModel();
38 
39  // first-order derivative
40  std::shared_ptr<fims_info::Information<TMB_FIMS_FIRST_ORDER>> d1 =
41  fims_info::Information<TMB_FIMS_FIRST_ORDER>::GetInstance();
42  d1->CreateModel();
43 
44  // second-order derivative
45  std::shared_ptr<fims_info::Information<TMB_FIMS_SECOND_ORDER>> d2 =
46  fims_info::Information<TMB_FIMS_SECOND_ORDER>::GetInstance();
47  d2->CreateModel();
48 
49  // third-order derivative
50  std::shared_ptr<fims_info::Information<TMB_FIMS_THIRD_ORDER>> d3 =
51  fims_info::Information<TMB_FIMS_THIRD_ORDER>::GetInstance();
52  d3->CreateModel();
53 
54  return true;
55 }
