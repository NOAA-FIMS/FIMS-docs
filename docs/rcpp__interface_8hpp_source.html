92std::string finalize_fims(Rcpp::NumericVector par, Rcpp::Function fn, Rcpp::Function gr) {
93
94 std::shared_ptr<fims_info::Information < double>> information =
95 fims_info::Information<double>::GetInstance();
96
97 std::shared_ptr<fims_model::Model < double>> model =
98 fims_model::Model<double>::GetInstance();
99 for (size_t i = 0; i < information->fixed_effects_parameters.size(); i++) {
100 *information->fixed_effects_parameters[i] = par[i];
101 }
102
103 bool reporting = model->do_tmb_reporting;
104 model->do_tmb_reporting = false;
105 model->Evaluate();
106
107 Rcpp::Function f = Rcpp::as<Rcpp::Function>(fn);
108 Rcpp::Function g = Rcpp::as<Rcpp::Function>(gr);
109 double val = Rcpp::as<double>(f(par));
110 Rcpp::NumericVector grad = Rcpp::as<Rcpp::NumericVector>(g(par));
111 double maxgc = -999;
112 for (R_xlen_t i = 0; i < grad.size(); i++) {
113 if (std::fabs(grad[i]) > maxgc) {
114 maxgc = std::fabs(grad[i]);
115 }
116 }
117
118 for (size_t i = 0; i < FIMSRcppInterfaceBase::fims_interface_objects.size();
119 i++) {
120 FIMSRcppInterfaceBase::fims_interface_objects[i]->finalize();
121 }
122 std::string ret;
123 auto now = std::chrono::system_clock::now();
124 std::time_t now_time = std::chrono::system_clock::to_time_t(now);
125 std::string ctime_no_newline = strtok(ctime(&now_time), "\n");
126 std::shared_ptr<fims_info::Information < double>> info =
127 fims_info::Information<double>::GetInstance();
128 std::stringstream ss;
129 ss << "{\n";
130 ss << "\"timestamp\": \"" << ctime_no_newline << "\",\n";
131 ss << "\"nyears\":" << info->nyears << ",\n";
132 ss << "\"nseasons\":" << info->nseasons << ",\n";
133 ss << "\"nages\":" << info->nages << ",\n";
134 ss << "\"objective_function_value\": " << val << ",\n";
135 ss << "\"max_gradient_component\": " << maxgc << ",\n";
136 ss << "\"final_gradient\": [";
137 if (grad.size() > 0) {
138 for (R_xlen_t i = 0; i < grad.size() - 1; i++) {
139 ss << grad[i] << ", ";
140 }
141 ss << grad[grad.size() - 1] << "],\n";
142 } else {
143 ss << "],";
144 }
145 ss << "\"modules\" : [\n";
146 size_t length = FIMSRcppInterfaceBase::fims_interface_objects.size();
147 if (length > 0) {
148 for (size_t i = 0; i < length - 1; i++) {
149 ss << FIMSRcppInterfaceBase::fims_interface_objects[i]->to_json() << ",\n";
150 }
151
152 ss << FIMSRcppInterfaceBase::fims_interface_objects[length - 1]->to_json() << "\n]\n}";
153
154 } else {
155 ss << "\n]\n}";
156 }
157
158 ret = fims::JsonParser::PrettyFormatJSON(ss.str());
159 model->do_tmb_reporting = reporting;
160 return ret;
161}

401std::string escapeQuotes(const std::string& input) {
402 std::string result = input;
403 std::string search = "\"";
404 std::string replace = "\\\"";
405
406 // Find each occurrence of `"` and replace it with `\"`
407 size_t pos = result.find(search);
408 while (pos != std::string::npos) {
409 result.replace(pos, search.size(), replace);
410 pos = result.find(search, pos + replace.size()); // Move past the replaced position
411 }
412 return result;
413}

418void log_error(std::string log_entry) {
419 std::stringstream ss;
420 ss << "capture.output(traceback(4))";
421 SEXP expression, result;
422 ParseStatus status;
423
424 PROTECT(expression = R_ParseVector(Rf_mkString(ss.str().c_str()), 1, &status, R_NilValue));
425 if (status != PARSE_OK) {
426 Rcpp::Rcout << "Error parsing expression" << std::endl;
427 UNPROTECT(1);
428 }
429 Rcpp::Rcout << "before call.";
430 PROTECT(result = Rf_eval(VECTOR_ELT(expression, 0), R_GlobalEnv));
431 Rcpp::Rcout << "after call.";
432 UNPROTECT(2);
433 std::stringstream ss_ret;
434 ss_ret << "traceback: ";
435 for (int j = 0; j < LENGTH(result); j++) {
436 std::string str(CHAR(STRING_ELT(result, j)));
437 ss_ret << escapeQuotes(str) << "\\n";
438 }
439
440 std::string ret = ss_ret.str(); //"find error";//Rcpp::as<std::string>(result);
441
442 fims::FIMSLog::fims_log->error_message(log_entry, -1, "R_env", ret.c_str());
443}
