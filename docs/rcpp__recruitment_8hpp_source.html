109  virtual double evaluate_nll() {

112  NLL.log_sigma_recruit = this->log_sigma_recruit.value;

115  recruit_bias_adjustment.size()); // Vector from TMB

117  NLL.recruit_deviations[i] = deviations[i];
118  NLL.recruit_bias_adjustment[i] = recruit_bias_adjustment[i];
119  }
120  FIMS_LOG << "Rec devs being passed to C++ are " << deviations << std::endl;
121 
122  Rcout << "Rec bias adj being passed to C++ are " << recruit_bias_adjustment
123  << std::endl;

125  NLL.use_recruit_bias_adjustment = this->use_bias_correction;
126  NLL.estimate_recruit_deviations = this->estimate_deviations;
