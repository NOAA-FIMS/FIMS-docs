342  Type CalculateSBPR0() {
343  std::vector<Type> numbers_spr(this->nages, 1.0);
344  Type phi_0 = 0.0;
345  phi_0 += numbers_spr[0] * this->proportion_female *
346  this->proportion_mature_at_age[0] *
347  this->growth->evaluate(ages[0]);
348  for (size_t a = 1; a < (this->nages - 1); a++) {
349  numbers_spr[a] = numbers_spr[a - 1] * fims::exp(-this->M[a]);
350  phi_0 += numbers_spr[a] * this->proportion_female *
351  this->proportion_mature_at_age[a] *
352  this->growth->evaluate(ages[a]);
353  }
354 
355  numbers_spr[this->nages - 1] =
356  (numbers_spr[nages - 2] * fims::exp(-this->M[nages - 2])) /
357  (1 - fims::exp(-this->M[this->nages - 1]));
358  phi_0 += numbers_spr[this->nages - 1] * this->proportion_female *
359  this->proportion_mature_at_age[this->nages - 1] *
360  this->growth->evaluate(ages[this->nages - 1]);
361  return phi_0;
362  }

535  void Evaluate() {
536  /*
537  Sets derived vectors to zero
538  Performs parameters transformations
539  Sets recruitment deviations to mean 0, and then adds bias adjustment.
540  */
541  Prepare();
542  /*
543  start at year=0, age=0;
544  here year 0 is the estimated initial stock structure and age 0 are recruits
545  loops start at zero with if statements inside to specify unique code for
546  initial structure and recruitment 0 loops. Could also have started loops at
547  1 with initial structure and recruitment setup outside the loops.
548 
549  year loop is extended to <= nyears because SSB is calculted as the start of
550  the year value and by extending one extra year we get estimates of the
551  population structure at the end of the final year. An alternative approach
552  would be to keep initial numbers at age in it's own vector and each year to
553  include the population structure at the end of the year. This is likely a
554  null point given that we are planning to modify to an event/stanza based
555  structure in later milestones which will elimitate this confusion by
556  explicity referencing the exact date (or period of averaging) at which any
557  calculation or output is being made.
558  */
559  for (size_t y = 0; y <= this->nyears; y++) {
560  for (size_t a = 0; a < this->nages; a++) {
561  /*
562  index naming defines the dimensional folding structure
563  i.e. index_ya is referencing folding over years and ages.
564  */
565  size_t index_ya = y * this->nages + a;
566  /*
567  Mortality rates are not estimated in the final year which is
568  used to show expected stock structure at the end of the model period.
569  This is because biomass in year i represents biomass at the start of
570  the year.
571  Should we add complexity to track more values such as start,
572  mid, and end biomass in all years where, start biomass=end biomass of
573  the previous year? Referenced above, this is probably not worth
574  exploring as later milestone changes will eliminate this confusion.
575  */
576  if (y < this->nyears) {
577  /*
578  First thing we need is total mortality aggregated across all fleets
579  to inform the subsequent catch and change in numbers at age
580  calculations. This is only calculated for years < nyears as these are
581  the model estimated years with data. The year loop extends to
582  y=nyears so that population numbers at age and SSB can be calculated
583  at the end of the last year of the model
584  */
585  CalculateMortality(index_ya, y, a);
586  }
587  CalculateMaturityAA(index_ya, a);
588  /* if statements needed because some quantities are only needed
589  for the first year and/or age, so these steps are included here.
590  */
591  if (y == 0) {
592  // Initial numbers at age is a user input or estimated parameter
593  // vector.
594  CalculateInitialNumbersAA(index_ya, a);
595 
596  if (a == 0) {
597  // this->numbers_at_age[index_ya] = this->recruitment->rzero;
598  this->unfished_numbers_at_age[index_ya] =
599  fims::exp(this->recruitment->log_rzero);
600  } else {
601  CalculateUnfishedNumbersAA(index_ya, a - 1, a);
602  }
603 
604  /*
605  Fished and unfished biomass vectors are summing biomass at
606  age across ages.
607  */
608 
609  CalculateBiomass(index_ya, y, a);
610 
611  CalculateUnfishedBiomass(index_ya, y, a);
612 
613  /*
614  Fished and unfished spawning biomass vectors are summing biomass at
615  age across ages to allow calculation of recruitment in the next year.
616  */
617 
618  CalculateSpawningBiomass(index_ya, y, a);
619 
620  CalculateUnfishedSpawningBiomass(index_ya, y, a);
621 
622  /*
623  Expected recruitment in year 0 is numbers at age 0 in year 0.
624  */
625 
626  this->expected_recruitment[index_ya] = this->numbers_at_age[index_ya];
627 
628  } else {
629  if (a == 0) {
630  // Set the nrecruits for age a=0 year y (use pointers instead of
631  // functional returns) assuming fecundity = 1 and 50:50 sex ratio
632  FIMS_LOG << "Recruitment: " << std::endl;
633  CalculateRecruitment(index_ya, y);
634  this->unfished_numbers_at_age[index_ya] =
635  fims::exp(this->recruitment->log_rzero);
636 
637  } else {
638  size_t index_ya2 = (y - 1) * nages + (a - 1);
639  CalculateNumbersAA(index_ya, index_ya2, a);
640  CalculateUnfishedNumbersAA(index_ya, index_ya2, a);
641  }
642  CalculateBiomass(index_ya, y, a);
643  CalculateSpawningBiomass(index_ya, y, a);
644 
645  FIMS_LOG << "index ya: " << index_ya << std::endl;
646  CalculateUnfishedBiomass(index_ya, y, a);
647  CalculateUnfishedSpawningBiomass(index_ya, y, a);
648  }
649 
650  /*
651  Here composition, total catch, and index values are calculated for all
652  years with reference data. They are not calculated for y=nyears as there
653  is this is just to get final population structure at the end of the
654  terminal year.
655  */
656  if (y < this->nyears) {
657  FIMS_LOG << index_ya << std::endl;
658  CalculateCatchNumbersAA(index_ya, y, a);
659 
660  CalculateCatchWeightAA(y, a);
661  FIMS_LOG << "year " << y << " and age " << a << std::endl;
662  CalculateCatch(y, a);
663  CalculateIndex(index_ya, y, a);
664  }
665  }
666  }
667  FIMS_LOG << "NAA\n";
668  for (int i = 0; i < nyears; i++) {
669  for (int j = 0; j < nages; j++) {
670  FIMS_LOG << numbers_at_age[i * nages + j] << "\t";
671  }
672  FIMS_LOG << "\n";
673  }
674

676  for (size_t fleet_ = 0; fleet_ < this->nfleets; fleet_++) {
677  FIMS_LOG << "Fleet " << fleet_ + 1 << "\n";
678  for (int i = 0; i < nyears; i++) {
679  for (int j = 0; j < nages; j++) {
680  FIMS_LOG << fleets[fleet_]->catch_numbers_at_age[i * nages + j]
681  << "\t";
682  }
683  FIMS_LOG << "\n";
684  }
685  }

690  // REPORT_F(int(this->nages), of); //REPORT error: call of overloaded is
