368  void CalculateCatchNumbersAA(int index_ya, int year, int age)
369  {
370  for (size_t fleet_ = 0; fleet_ < this->nfleets; fleet_++)
371  {
372  int index_yaf = year * this->nages * this->nfleets + age * this->nfleets + fleet_;
373  int index_yf = year * this->nfleets + fleet_; // index by fleet and years to dimension fold
374  // make an intermediate value in order to set multiple members (of
375  // current and fleet objects) to that value.
376  Type catch_; // catch_ is used to avoid using the c++ keyword catch
377  // Baranov Catch Equation
378  catch_ =
379  (this->fleets[fleet_]->Fmort[year] *
380  this->fleets[fleet_]->selectivity->evaluate(age)) /
381  this->mortality_Z[index_ya] *
382  this->numbers_at_age[index_ya] *
383  (1 - exp(-(this->mortality_Z[index_ya])));
384  this->catch_numbers_at_age[index_yaf] += catch_;
385  // catch_numbers_at_age for the fleet module has different
386  // dimensions (year/age, not year/fleet/age)
387  fleets[fleet_]->catch_numbers_at_age[index_ya] += catch_;
388  }

426  void Evaluate()
427  {
428 
429  /*
430  Sets derived vectors to zero
431  Performs parameters transformations
432  Sets recruitment deviations to mean 0, and then adds bias adjustment.
433  */
434  Prepare();
435  /*
436  start at year=0, age=0;
437  here year 0 is the estimated initial stock structure and age 0 are recruits
438  loops start at zero with if statements inside to specify unique code for
439  initial structure and recruitment 0 loops. Could also have started loops at
440  1 with initial structure and recruitment setup outside the loops.
441 
442  year loop is extended to <= nyears because SSB is calculted as the start of the year
443  value and by extending one extra year we get estimates of the population structure
444  at the end of the final year. An alternative approach would be to keep initial numbers
445  at age in it's own vector and each year to include the population structure at the end of
446  the year. This is likely a null point given that we are planning to modify to an event/stanza
447  based structure in later milestones which will elimitate this confusion by explicity
448  referencing the exact date (or period of averaging) at which any calculation or output is
449  being made.
450  */
451  for (size_t y = 0; y <= this->nyears; y++)
452  {
453  for (size_t a = 0; a < this->nages; a++)
454  {
455  /*
456  index naming defines the dimensional folding structure
457  i.e. index_ya is referencing folding over years and ages.
458  */
459  int index_ya = y * this->nages + a;
460  /*
461  Mortality rates are not estimated in the final year which is
462  used to show expected stock structure at the end of the model period.
463  This is because biomass in year i represents biomass at the start of
464  the year.
465  Should we add complexity to track more values such as start,
466  mid, and end biomass in all years where, start biomass=end biomass of the
467  previous year? Referenced above, this is probably not worth exploring as
468  later milestone changes will eliminate this confusion.
469  */
470  if (y < this->nyears)
471  {
472  /*
473  First thing we need is total mortality aggregated across all fleets
474  to inform the subsequent catch and change in numbers at age calculations.
475  This is only calculated for years < nyears as these are the model estimated
476  years with data. The year loop extends to y=nyears so that population numbers
477  at age and SSB can be calculated at the end of the last year of the model
478  */
479  CalculateMortality(index_ya, y, a);
480  }
481  CalculateMaturityAA(index_ya, a);
482  /* if statements needed because some quantities are only needed
483  for the first year and/or age, so these steps are included here.
484  */
485  if (y == 0)
486  {
487  // Initial numbers at age is a user input or estimated parameter vector.
488  CalculateInitialNumbersAA(index_ya, a);
489  if (a == 0)
490  {
491  this->unfished_numbers_at_age[index_ya] =
492  this->recruitment->rzero;
493  }
494  else
495  {
496  // CalculateUnfishedNumbersAA(index_ya, a);
497  // CalculateUnfishedNumbersAA(index_ya, index_ya-1);
498  CalculateUnfishedNumbersAA(index_ya, a-1, a);
499  }
500  /*
501  Fished and unfished spawning biomass vectors are summing biomass at age
502  across ages to allow calculation of recruitment in the next year.
503  */
504  CalculateSpawningBiomass(index_ya, y, a);
505  CalculateUnfishedSpawningBiomass(index_ya, y, a);
506  }
507  else
508  {
509  if (a == 0)
510  {
511  // Set the nrecruits for age a=0 year y (use pointers instead of functional returns)
512  // assuming fecundity = 1 and 50:50 sex ratio
513  CalculateRecruitment(index_ya, y);
514  this->unfished_numbers_at_age[index_ya] =
515  this->recruitment->rzero;
516  }
517  else
518  {
519  int index_ya2 = (y - 1) * nages + (a - 1);
520  CalculateNumbersAA(index_ya, index_ya2, a);
521  CalculateUnfishedNumbersAA(index_ya, index_ya2, a);
522  }
523  CalculateSpawningBiomass(index_ya, y, a);
524  CalculateUnfishedSpawningBiomass(index_ya, y, a);
525  }
526 
527  /*
528  Here composition, total catch, and index values are calculated for all years with reference
529  data. They are not calculated for y=nyears as there is this is just to get final population
530  structure at the end of the terminal year.
531  */
532  if (y < this->nyears)
533  {
534  CalculateCatchNumbersAA(index_ya, y, a);
535  CalculateCatchWeightAA(y, a);
536  CalculateCatch(y, a);
537  CalculateIndex(index_ya, y, a);
538  }
539  }
540  }
541  }
