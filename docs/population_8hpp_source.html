353  Type CalculateSBPR0() {
354  std::vector<Type> numbers_spr(this->nages, 1.0);
355  Type phi_0 = 0.0;
356  phi_0 += numbers_spr[0] * this->proportion_female *
357  this->proportion_mature_at_age[0] *
358  this->growth->evaluate(ages[0]);
359  for (size_t a = 1; a < (this->nages - 1); a++) {
360  numbers_spr[a] = numbers_spr[a - 1] * fims_math::exp(-this->M[a]);
361  phi_0 += numbers_spr[a] * this->proportion_female *
362  this->proportion_mature_at_age[a] *
363  this->growth->evaluate(ages[a]);
364  }
365 
366  numbers_spr[this->nages - 1] =
367  (numbers_spr[nages - 2] * fims_math::exp(-this->M[nages - 2])) /
368  (1 - fims_math::exp(-this->M[this->nages - 1]));
369  phi_0 += numbers_spr[this->nages - 1] * this->proportion_female *
370  this->proportion_mature_at_age[this->nages - 1] *
371  this->growth->evaluate(ages[this->nages - 1]);
372  return phi_0;
373  }

552  void Evaluate() {
553  /*
554  Sets derived vectors to zero
555  Performs parameters transformations
556  Sets recruitment deviations to mean 0.
557  */
558  Prepare();
559  /*
560  start at year=0, age=0;
561  here year 0 is the estimated initial stock structure and age 0 are recruits
562  loops start at zero with if statements inside to specify unique code for
563  initial structure and recruitment 0 loops. Could also have started loops at
564  1 with initial structure and recruitment setup outside the loops.
565 
566  year loop is extended to <= nyears because SSB is calculted as the start of
567  the year value and by extending one extra year we get estimates of the
568  population structure at the end of the final year. An alternative approach
569  would be to keep initial numbers at age in it's own vector and each year to
570  include the population structure at the end of the year. This is likely a
571  null point given that we are planning to modify to an event/stanza based
572  structure in later milestones which will elimitate this confusion by
573  explicity referencing the exact date (or period of averaging) at which any
574  calculation or output is being made.
575  */
576  for (size_t y = 0; y <= this->nyears; y++) {
577  for (size_t a = 0; a < this->nages; a++) {
578  /*
579  index naming defines the dimensional folding structure
580  i.e. i_age_year is referencing folding over years and ages.
581  */
582  size_t i_age_year = y * this->nages + a;
583  /*
584  Mortality rates are not estimated in the final year which is
585  used to show expected stock structure at the end of the model period.
586  This is because biomass in year i represents biomass at the start of
587  the year.
588  Should we add complexity to track more values such as start,
589  mid, and end biomass in all years where, start biomass=end biomass of
590  the previous year? Referenced above, this is probably not worth
591  exploring as later milestone changes will eliminate this confusion.
592  */
593  if (y < this->nyears) {
594  /*
595  First thing we need is total mortality aggregated across all fleets
596  to inform the subsequent catch and change in numbers at age
597  calculations. This is only calculated for years < nyears as these are
598  the model estimated years with data. The year loop extends to
599  y=nyears so that population numbers at age and SSB can be calculated
600  at the end of the last year of the model
601  */
602  CalculateMortality(i_age_year, y, a);
603  }
604  CalculateMaturityAA(i_age_year, a);
605  /* if statements needed because some quantities are only needed
606  for the first year and/or age, so these steps are included here.
607  */
608  if (y == 0) {
609  // Initial numbers at age is a user input or estimated parameter
610  // vector.
611  CalculateInitialNumbersAA(i_age_year, a);
612 
613  if (a == 0) {
614  // this->numbers_at_age[i_age_year] = this->recruitment->rzero;
615  this->unfished_numbers_at_age[i_age_year] =
616  fims_math::exp(this->recruitment->log_rzero);
617  } else {
618  CalculateUnfishedNumbersAA(i_age_year, a - 1, a);
619  }
620 
621  /*
622  Fished and unfished biomass vectors are summing biomass at
623  age across ages.
624  */
625 
626  CalculateBiomass(i_age_year, y, a);
627 
628  CalculateUnfishedBiomass(i_age_year, y, a);
629 
630  /*
631  Fished and unfished spawning biomass vectors are summing biomass at
632  age across ages to allow calculation of recruitment in the next year.
633  */
634 
635  CalculateSpawningBiomass(i_age_year, y, a);
636 
637  CalculateUnfishedSpawningBiomass(i_age_year, y, a);
638 
639  /*
640  Expected recruitment in year 0 is numbers at age 0 in year 0.
641  */
642 
643  this->expected_recruitment[i_age_year] =
644  this->numbers_at_age[i_age_year];
645 
646  } else {
647  if (a == 0) {
648  // Set the nrecruits for age a=0 year y (use pointers instead of
649  // functional returns) assuming fecundity = 1 and 50:50 sex ratio
650  POPULATION_LOG << "Recruitment: " << std::endl;
651  CalculateRecruitment(i_age_year, y, y);
652  this->unfished_numbers_at_age[i_age_year] =
653  fims_math::exp(this->recruitment->log_rzero);
654 
655  } else {
656  size_t i_agem1_yearm1 = (y - 1) * nages + (a - 1);
657  CalculateNumbersAA(i_age_year, i_agem1_yearm1, a);
658  CalculateUnfishedNumbersAA(i_age_year, i_agem1_yearm1, a);
659  }
660  CalculateBiomass(i_age_year, y, a);
661  CalculateSpawningBiomass(i_age_year, y, a);
662 
663  POPULATION_LOG << "index age year: " << i_age_year << std::endl;
664  CalculateUnfishedBiomass(i_age_year, y, a);
665  CalculateUnfishedSpawningBiomass(i_age_year, y, a);
666  }
667 
668  /*
669  Here composition, total catch, and index values are calculated for all
670  years with reference data. They are not calculated for y=nyears as there
671  is this is just to get final population structure at the end of the
672  terminal year.
673  */
674  if (y < this->nyears) {
