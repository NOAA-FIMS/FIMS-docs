489  void Evaluate() {
490  /*
491  Sets derived vectors to zero
492  Performs parameters transformations
493  Sets recruitment deviations to mean 0.
494  */
495  Prepare();
496  /*
497  start at year=0, age=0;
498  here year 0 is the estimated initial stock structure and age 0 are recruits
499  loops start at zero with if statements inside to specify unique code for
500  initial structure and recruitment 0 loops. Could also have started loops at
501  1 with initial structure and recruitment setup outside the loops.
502 
503  year loop is extended to <= nyears because SSB is calculted as the start of
504  the year value and by extending one extra year we get estimates of the
505  population structure at the end of the final year. An alternative approach
506  would be to keep initial numbers at age in it's own vector and each year to
507  include the population structure at the end of the year. This is likely a
508  null point given that we are planning to modify to an event/stanza based
509  structure in later milestones which will elimitate this confusion by
510  explicity referencing the exact date (or period of averaging) at which any
511  calculation or output is being made.
512  */
513  for (size_t y = 0; y <= this->nyears; y++) {
514  for (size_t a = 0; a < this->nages; a++) {
515  /*
516  index naming defines the dimensional folding structure
517  i.e. i_age_year is referencing folding over years and ages.
518  */
519  size_t i_age_year = y * this->nages + a;
520  /*
521  Mortality rates are not estimated in the final year which is
522  used to show expected stock structure at the end of the model period.
523  This is because biomass in year i represents biomass at the start of
524  the year.
525  Should we add complexity to track more values such as start,
526  mid, and end biomass in all years where, start biomass=end biomass of
527  the previous year? Referenced above, this is probably not worth
528  exploring as later milestone changes will eliminate this confusion.
529  */
530  if (y < this->nyears) {
531  /*
532  First thing we need is total mortality aggregated across all fleets
533  to inform the subsequent catch and change in numbers at age
534  calculations. This is only calculated for years < nyears as these are
535  the model estimated years with data. The year loop extends to
536  y=nyears so that population numbers at age and SSB can be calculated
537  at the end of the last year of the model
538  */
539  CalculateMortality(i_age_year, y, a);
540  }
541  CalculateMaturityAA(i_age_year, a);
542  /* if statements needed because some quantities are only needed
543  for the first year and/or age, so these steps are included here.
544  */
545  if (y == 0) {
546  // Initial numbers at age is a user input or estimated parameter
547  // vector.
548  CalculateInitialNumbersAA(i_age_year, a);
549 
550  if (a == 0) {
551  // this->numbers_at_age[i_age_year] = this->recruitment->rzero;
552  this->unfished_numbers_at_age[i_age_year] =
553  fims_math::exp(this->recruitment->log_rzero[0]);
554  } else {
555  CalculateUnfishedNumbersAA(i_age_year, a - 1, a);
556  }
557 
558  /*
559  Fished and unfished biomass vectors are summing biomass at
560  age across ages.
561  */
562 
563  CalculateBiomass(i_age_year, y, a);
564 
565  CalculateUnfishedBiomass(i_age_year, y, a);
566 
567  /*
568  Fished and unfished spawning biomass vectors are summing biomass at
569  age across ages to allow calculation of recruitment in the next year.
570  */
571 
572  CalculateSpawningBiomass(i_age_year, y, a);
573 
574  CalculateUnfishedSpawningBiomass(i_age_year, y, a);
575 
576  /*
577  Expected recruitment in year 0 is numbers at age 0 in year 0.
578  */
579 
580  this->expected_recruitment[i_age_year] =
581  this->numbers_at_age[i_age_year];
582 
583  } else {
584  if (a == 0) {
585  // Set the nrecruits for age a=0 year y (use pointers instead of
586  // functional returns) assuming fecundity = 1 and 50:50 sex ratio
587  CalculateRecruitment(i_age_year, y, y);
588  this->unfished_numbers_at_age[i_age_year] =
589  fims_math::exp(this->recruitment->log_rzero[0]);
590 
591  } else {
592  size_t i_agem1_yearm1 = (y - 1) * nages + (a - 1);
593  CalculateNumbersAA(i_age_year, i_agem1_yearm1, a);
594  CalculateUnfishedNumbersAA(i_age_year, i_agem1_yearm1, a);
595  }
596  CalculateBiomass(i_age_year, y, a);
597  CalculateSpawningBiomass(i_age_year, y, a);
598 
599  CalculateUnfishedBiomass(i_age_year, y, a);
600  CalculateUnfishedSpawningBiomass(i_age_year, y, a);
601  }
602 
603  /*
604  Here composition, total catch, and index values are calculated for all
605  years with reference data. They are not calculated for y=nyears as there
606  is this is just to get final population structure at the end of the
607  terminal year.
608  */
609  if (y < this->nyears) {
610  CalculateCatchNumbersAA(i_age_year, y, a);
611 
612  CalculateCatchWeightAA(y, a);
613  CalculateCatch(y, a);
614  CalculateIndex(i_age_year, y, a);
615  }
616  }
617  }
