62  const Type Evaluate() {
63  // jnll = negative-log-likelihood (the objective function)
64  Type jnll = 0.0;
65  Type rec_nll = 0.0; // recrutiment nll
66  Type age_comp_nll = 0.0; // age composition nll
67  Type index_nll = 0.0; // survey and fishery cacth nll
68 
69  int n_fleets = fims_information->fleets.size();
70  int n_pops = fims_information->populations.size();
71 
72 // Create vector lists to store output for reporting
73 #ifdef TMB_MODEL
74  // vector< vector<Type> > creates a nested vector structure where
75  // each vector can be a different dimension. Does not work with ADREPORT
76  // fleets
77  vector<vector<Type> > exp_index(n_fleets);
78  vector<vector<Type> > exp_catch(n_fleets);
79  vector<vector<Type> > cnaa(n_fleets);
80  vector<vector<Type> > cwaa(n_fleets);
81  vector<vector<Type> > F_mort(n_fleets);
82  // populations
83  vector<vector<Type> > naa(n_pops);
84  vector<vector<Type> > ssb(n_pops);
85  vector<vector<Type> > biomass(n_pops);
86  vector<vector<Type> > log_recruit_dev(n_pops);
87  vector<vector<Type> > recruitment(n_pops);
88  vector<vector<Type> > M(n_pops);
89 
90 #endif
91 
92  // Loop over populations, evaluate, and sum up the recruitment likelihood
93  // component
94 
95  typename fims_info::Information<Type>::population_iterator it;
96  MODEL_LOG << "Evaluating expected values and summing recruitment nlls for "
97  << this->fims_information->populations.size() << " populations."
98  << std::endl;
99 
100  for (it = this->fims_information->populations.begin();
101  it != this->fims_information->populations.end(); ++it) {
102  //(*it).second points to the Population module
103  MODEL_LOG << "Setting up pointer to population " << (*it).second->id
104  << "." << std::endl;
105  // Prepare recruitment
106  (*it).second->recruitment->Prepare();
107  MODEL_LOG << "Recruitment for population successfully prepared"
108  << std::endl;
109 // link to TMB objective function
110 #ifdef TMB_MODEL
111  (*it).second->of = this->of;
112 #endif
113  // Evaluate population
114  (*it).second->Evaluate();
115  // Recrtuiment negative log-likelihood
116  rec_nll += (*it).second->recruitment->evaluate_nll();
117  MODEL_LOG << "Recruitment negative log-likelihood is: " << rec_nll
118  << std::endl;
119  }
120  MODEL_LOG << "All populations successfully evaluated." << std::endl;
121 
122  // Loop over fleets/surveys, and sum up age comp and index nlls
123 
124  typename fims_info::Information<Type>::fleet_iterator jt;
125  MODEL_LOG << "Evaluating expected values and summing nlls for "
126  << this->fims_information->fleets.size() << " fleets."
127  << std::endl;
128 
129  for (jt = this->fims_information->fleets.begin();
130  jt != this->fims_information->fleets.end(); ++jt) {
131  //(*jt).second points to each individual Fleet module

133  (*jt).second->of = this->of;
134 #endif
135  MODEL_LOG << "Setting up pointer to fleet " << (*jt).second->id << "."

137  age_comp_nll += (*jt).second->evaluate_age_comp_nll();
138  MODEL_LOG << "Sum of survey and age comp negative log-likelihood is: "

140  index_nll += (*jt).second->evaluate_index_nll();
141  }
142  MODEL_LOG << "All fleets successfully evaluated." << std::endl;
143  // Loop over populations and fleets/surveys and fill in reporting
144 
145  // initiate population index for structuring report out objects
146  int pop_idx = 0;
147  for (it = this->fims_information->populations.begin();
148  it != this->fims_information->populations.end(); ++it) {
149 #ifdef TMB_MODEL
150  naa(pop_idx) = vector<Type>((*it).second->numbers_at_age);
151  ssb(pop_idx) = vector<Type>((*it).second->spawning_biomass);
152  log_recruit_dev(pop_idx) =
153  vector<Type>((*it).second->recruitment->log_recruit_devs);
154  recruitment(pop_idx) = vector<Type>((*it).second->expected_recruitment);
155  biomass(pop_idx) = vector<Type>((*it).second->biomass);
156  M(pop_idx) = vector<Type>((*it).second->M);
157 #endif
158  pop_idx += 1;
159  }
160 
161  // initiate fleet index for structuring report out objects
162  int fleet_idx = 0;
163  for (jt = this->fims_information->fleets.begin();
164  jt != this->fims_information->fleets.end(); ++jt) {
165 #ifdef TMB_MODEL

167  exp_catch(fleet_idx) = (*jt).second->expected_catch;
168  F_mort(fleet_idx) = (*jt).second->Fmort;
169  cnaa(fleet_idx) = (*jt).second->catch_numbers_at_age;

172  fleet_idx += 1;
173  }
