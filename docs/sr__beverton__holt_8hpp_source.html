48  Type recruits;

54  // Transform input parameters
55  steep = fims_math::inv_logit(steep_lo, steep_hi, this->logit_steep);
56  rzero = fims_math::exp(this->log_rzero);
57 
58  recruits = (0.8 * rzero * steep * spawners) /
59  (0.2 * phi_0 * rzero * (1.0 - steep) + spawners * (steep - 0.2));
