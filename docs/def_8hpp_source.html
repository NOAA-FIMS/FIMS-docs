574  void WriteAtExit(int sig) {
575 
576  std::string signal_error = "NA";
577  switch (sig) {
578  case SIGSEGV:
579  signal_error = "Invalid memory access (segmentation fault)";
580  break;
581  case SIGINT:
582  signal_error = "External interrupt, possibly initiated by the user.";
583  break;
584  case SIGABRT:
585  signal_error = "Abnormal termination condition, possible call to std::abort.";
586  break;
587  case SIGFPE:
588  signal_error = "Erroneous arithmetic operation.";
589  break;
590  case SIGILL:
591  signal_error = "Invalid program image or invalid instruction";
592  break;
593  case SIGTERM:
594  signal_error = "Termination request, sent to the program.";
595  break;
596  default:
597  signal_error = "Unknown signal thrown";
598 
599  }
600 
601  FIMSLog::fims_log->error_message(signal_error, -999, "?", "?");
602 
603 
604  if (FIMSLog::fims_log->write_on_exit) {
605 
606  std::ofstream log(FIMSLog::fims_log->get_path());
607  log << FIMSLog::fims_log->get_log();
608  log.close();
609  }
610  std::signal(sig, SIG_DFL);
611  raise(sig);
612  }
