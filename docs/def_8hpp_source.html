587void WriteAtExit(int sig) {
588 std::string signal_error = "NA";
589 switch (sig) {
590 case SIGSEGV:
591 signal_error = "Invalid memory access (segmentation fault)";
592 break;
593 case SIGINT:
594 signal_error = "External interrupt, possibly initiated by the user.";
595 break;
596 case SIGABRT:
597 signal_error =
598 "Abnormal termination condition, possible call to std::abort.";
599 break;
600 case SIGFPE:
601 signal_error = "Erroneous arithmetic operation.";
602 break;
603 case SIGILL:
604 signal_error = "Invalid program image or invalid instruction";
605 break;
606 case SIGTERM:
607 signal_error = "Termination request, sent to the program.";
608 break;
609 default:
610 signal_error = "Unknown signal thrown";
611 }
612
613 FIMSLog::fims_log->error_message(signal_error, -999, "?", "?");
614
615 if (FIMSLog::fims_log->write_on_exit) {
616 std::ofstream log(FIMSLog::fims_log->get_path());
617 log << FIMSLog::fims_log->get_log();
618 log.close();
619 }
620 std::signal(sig, SIG_DFL);
621 raise(sig);
622}
