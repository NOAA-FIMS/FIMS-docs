577 void WriteAtExit(int sig) {
578
579 std::string signal_error = "NA";
580 switch (sig) {
581 case SIGSEGV:
582 signal_error = "Invalid memory access (segmentation fault)";
583 break;
584 case SIGINT:
585 signal_error = "External interrupt, possibly initiated by the user.";
586 break;
587 case SIGABRT:
588 signal_error = "Abnormal termination condition, possible call to std::abort.";
589 break;
590 case SIGFPE:
591 signal_error = "Erroneous arithmetic operation.";
592 break;
593 case SIGILL:
594 signal_error = "Invalid program image or invalid instruction";
595 break;
596 case SIGTERM:
597 signal_error = "Termination request, sent to the program.";
598 break;
599 default:
600 signal_error = "Unknown signal thrown";
601
602 }
603
604 FIMSLog::fims_log->error_message(signal_error, -999, "?", "?");
605
606
607 if (FIMSLog::fims_log->write_on_exit) {
608
609 std::ofstream log(FIMSLog::fims_log->get_path());
610 log << FIMSLog::fims_log->get_log();
611 log.close();
612 }
613 std::signal(sig, SIG_DFL);
614 raise(sig);
615 }
